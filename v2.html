PERSONNALISEZ VOTRE SITE

Ecrire "background-color:" suivit de votre couleur de fond du site(blue,red ou green )
exemple:"background-color:blue"

Ecrire "color:" suivit de votre couleur de titre(purple,yellow ou pink )
exemple:"color:purple"

Ecrire "background-color:" suivit de votre couleur de fond du titre(white,beige ou firebrick )
exemple:"background-color:white"

Ecrire "color:" suivit de votre couleur de texte(maroon,white )
exemple:"color:white"

Ecrire "text-align:" suivit de votre emplacement de texte(left,center ou right )
exemple:"text-align:left"

Ecrire "font-size:" suivit de votre emplacement de texte(small,medium ou big )
exemple:"font-size:small"

Ecrire "src:" suivit de votre photo (cat,therock ou dog )
exemple:"src:therock"

Ecrire "margin:" suivit de votre emplacement de bouton (left ou right)
exemple:"margin:left"

Ecrire "border-radius:" suivit de votre emplacement de bouton (rectangle,rond ou ovale) exemple:"border-radius:rectangle"
